  <script>
    let html5QrCode;

    function startScan() {
      document.getElementById("result").innerText = "";

      html5QrCode = new Html5Qrcode("reader");

      html5QrCode.start(
        { facingMode: "environment" }, // back camera
        {
          fps: 10,
          qrbox: 250
        },
       qrCodeMessage => {
  html5QrCode.stop();

  // Save sheet link
  localStorage.setItem("sheetLink", qrCodeMessage);

  // Redirect back to bot
  window.location.href = "/bot.html";
},
        errorMessage => {
          // ignore scan errors
        }
      ).catch(err => {
        alert("Camera error: " + err);
      });
    }
  </script>
